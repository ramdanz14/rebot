SELECT kodetoko,typetoko24,IFNULL(jml_s4,0)  AS jml_S4,tgls4 FROM mstr_toko_today LEFT JOIN
(SELECT toko,COUNT(*) AS jml_s4,GROUP_CONCAT(DISTINCT DAY(pos_date) SEPARATOR "_") tgls4 FROM harian_2206  h WHERE shift='4' AND pos_date RLIKE '2022-06' GROUP BY toko) a ON kodetoko=toko WHERE typetoko24='Y' OR jml_s4>0;
